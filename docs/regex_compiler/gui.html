 25def _format_value(value) -> str:
 26    """Extract display value from State/Symbol objects."""
 27    attr = getattr(value, "value", None)
 28    if attr is not None:
 29        return str(attr)
 30    return str(value)

 33def fsm_to_networkx(fsm: FSMType):
 34    """Convert FSM to NetworkX MultiDiGraph. Renames $-prefixed states to s1, s2, etc."""
 35    G = nx.MultiDiGraph()
 36    state_mapping = {}
 37    counter = 1
 38    for s in fsm.STATES:
 39        label = _format_value(s)
 40        if label.startswith('$'):
 41            state_mapping[label] = f's{counter}'
 42            counter += 1
 43        else:
 44            state_mapping[label] = label
 45    for s in fsm.STATES:
 46        original_label = _format_value(s)
 47        display_label = state_mapping[original_label]
 48        G.add_node(display_label, initial=(s == fsm.INITIAL_STATE), accepting=(
 49            s in fsm.ACCEPTING_STATES))
 50    edge_aggregator = {}
 51    for (s, a), tgt in fsm.TRANSITIONS.items():
 52        s_label = state_mapping[_format_value(s)]
 53        symbol_label = _format_value(a)
 54        if symbol_label == "":
 55            symbol_label = "ε"
 56        if isinstance(tgt, (set, frozenset)):
 57            for t in tgt:
 58                t_label = state_mapping[_format_value(t)]
 59                edge_key = (s_label, t_label)
 60                if edge_key not in edge_aggregator:
 61                    edge_aggregator[edge_key] = []
 62                edge_aggregator[edge_key].append(symbol_label)
 63        else:
 64            t_label = state_mapping[_format_value(tgt)]
 65            edge_key = (s_label, t_label)
 66            if edge_key not in edge_aggregator:
 67                edge_aggregator[edge_key] = []
 68            edge_aggregator[edge_key].append(symbol_label)
 69
 70    for s, targets in getattr(fsm, "EPS_TRANSITIONS", {}).items():
 71        s_label = state_mapping[_format_value(s)]
 72        for t in targets:
 73            t_label = state_mapping[_format_value(t)]
 74            edge_key = (s_label, t_label)
 75            if edge_key not in edge_aggregator:
 76                edge_aggregator[edge_key] = []
 77            edge_aggregator[edge_key].append("ε")
 78    for (s_label, t_label), labels in edge_aggregator.items():
 79        combined_label = ",".join(sorted(set(labels)))
 80        G.add_edge(s_label, t_label, label=combined_label)
 81    return G

 84def draw_fsm_on_axes(ax: plt.axis, fsm: FSMType, layout_seed: int = 1):
 85    """Draw the FSM on the given matplotlib axes."""
 86    ax.clear()
 87    G = fsm_to_networkx(fsm)
 88    pos = nx.spring_layout(G, seed=layout_seed, k=1.5, iterations=50)
 89    initial_node = None
 90    accepting_nodes = []
 91    initial_accepting = []
 92    regular_nodes = []
 93    for node in G.nodes():
 94        is_initial = G.nodes[node].get('initial', False)
 95        is_accepting = G.nodes[node].get('accepting', False)
 96        if is_initial and is_accepting:
 97            initial_accepting.append(node)
 98        elif is_initial:
 99            initial_node = node
100        elif is_accepting:
101            accepting_nodes.append(node)
102        else:
103            regular_nodes.append(node)
104    if regular_nodes:
105        nx.draw_networkx_nodes(G, pos, nodelist=regular_nodes, node_color='lightblue',
106                               node_size=1200, ax=ax)
107    if initial_node:
108        nx.draw_networkx_nodes(G, pos, nodelist=[initial_node], node_color='lightgreen',
109                               node_size=1200, ax=ax, node_shape='s')
110    if accepting_nodes:
111        nx.draw_networkx_nodes(G, pos, nodelist=accepting_nodes, node_color='gold',
112                               node_size=1200, ax=ax, linewidths=3, edgecolors='orange')
113    if initial_accepting:
114        nx.draw_networkx_nodes(G, pos, nodelist=initial_accepting, node_color='lightgreen',
115                               node_size=1200, ax=ax, node_shape='s', linewidths=3, edgecolors='orange')
116    node_labels = {n: n for n in G.nodes}
117    nx.draw_networkx_labels(G, pos, labels=node_labels, ax=ax)
118    edge_labels = {(u, v): data.get("label", "")
119                   for u, v, data in G.edges(data=True)}
120    nx.draw_networkx_edges(G, pos, ax=ax, connectionstyle="arc3,rad=0.1",
121                           arrows=True, arrowsize=20, arrowstyle='->',
122                           node_size=1200,
123                           min_source_margin=15, min_target_margin=15)
124    nx.draw_networkx_edge_labels(G, pos, edge_labels=edge_labels, ax=ax)
125    ax.set_axis_off()
126    return ax

246    def safe_run(self, fn, *a, **kw) -> bool:
247        """Execute function with error handling and GUI feedback."""
248        try:
249            fn(*a, **kw)
250        except ParserError as e:
251            self.set_status("Parser error")
252            QMessageBox.critical(self, "Parser Error", str(e))
253            return False
254        except Exception as e:
255            self.set_status("Error (see console)")
256            error_msg = f"{type(e).__name__}: {str(e)}"
257            QMessageBox.critical(self, "Error", error_msg)
258            traceback.print_exc()
259            return False
260        return True

296    def on_convert_eps_to_nfa(self):
297        """Convert the current ε-NFA to an NFA."""
298        self.layout_seed = 1

311    def on_convert_nfa_to_dfa(self):
312        """Convert the current NFA to a DFA."""
313        self.layout_seed = 1
314        if self.current_fsm is None:
315            self.set_status("No automaton to convert")
316            return
317        if isinstance(self.current_fsm, DFA):
318            self.set_status("Current automaton is already a DFA")
319            return
320        if isinstance(self.current_fsm, EpsNFA):
321            self.set_status("Convert ε-NFA to NFA before creating a DFA")
322            return
323        try:
324            converted: DFA = convert_to_dfa(self.current_fsm)

327            traceback.print_exc()
328            return
329        self.set_status("Converted NFA to DFA")
330        self.set_current_fsm(converted)

332    def on_parse_regex(self):
333        """Parse regex and build ε-NFA."""
334        self.layout_seed = 1
335        txt = self.regex_text.toPlainText().strip()
336        if not txt:
337            self.set_status("Empty regex input")
338            return
339
340        def work():
341            ast: Expr = parse_regex(lex_regex(txt))
342            eps: EpsNFA = regex_to_epsnfa(ast)
343            self.set_current_fsm(eps)
344        if self.safe_run(work):
345            self.set_status("Parsed regex")
346
347    def on_parse_spec(self):
348        """Parse automaton specification and create FSM."""
349        self.layout_seed = 1
350        txt = self.spec_text.toPlainText().strip()
351        if not txt:
352            self.set_status("Empty spec input")
353            return
354
355        def work():
356            fsm: FSMType = parse_automaton(lex_automaton(txt))
357            self.set_current_fsm(fsm)
358        if self.safe_run(work):
359            self.set_status("Parsed spec")
360
361    def on_check_word(self):
362        """Test if current automaton accepts the input word."""
363        self.layout_seed = 1
364        if self.current_fsm is None:
365            self.acceptance_label.setText("No automaton loaded")
366            self.acceptance_label.setStyleSheet("color: red")
367            return
368
369        word = self.test_entry.text().strip()
370        try:
371            symbols: List[Symbol] = [Symbol(c) for c in word]
372            accepts: bool = self.current_fsm.accepts(symbols)
373            if accepts:
374                self.acceptance_label.setText("✓ ACCEPTED")
375                self.acceptance_label.setStyleSheet("color: green")
376            else:
377                self.acceptance_label.setText("✗ REJECTED")
378                self.acceptance_label.setStyleSheet("color: red")
379        except Exception as e:
380            self.acceptance_label.setText("Error")
381            self.acceptance_label.setStyleSheet("color: red")
382            traceback.print_exc()
